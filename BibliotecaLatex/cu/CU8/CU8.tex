	\begin{UseCase}{CU08}{Préstamo interbibliotecario}{
		Este caso de uso tiene como objetivo prestar un libro a otra biblioteca para satisfacer la necesidad de libros mediante el llenado de un formulario 
		
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Administrador}
		\UCitem{Propósito}{Prestar un libro de la biblioteca a otra biblioteca, se llenara un formulario donde se introducirá la biblioteca a la que se le prestara el libro y el ISBN libro, se verificara si el libro se puede prestar, y si se tiene convenio con la biblioteca.}
		\UCitem{Entradas}{ISBN del libro y nombre del la biblioteca}
		\UCitem{Salidas}{un mensaje del préstamo donde se especificara el libro que se presta, biblioteca a la que se le presta, fecha actual y fecha de devolución}
		\UCitem{Precondiciones}{ 
			\begin{itemize}
				\item El libro debe estar disponible para préstamo. 
				\item conexión a la base de datos.
				\item Se debe tener convenio con la biblioteca
			\end{itemize}
		}
		\UCitem{Postcondiciones}{El libro estará en la base de datos como prestado}
		\UCitem{Autor}{Rodriguez Ceravates Arturo y Monserrat Ceron Rodriguez .}
		\UCitem{Reviso}{Abidal y Diego}
	\end{UseCase}
