	\begin{UseCase}{CU08}{Préstamo interbibliotecario}{
		El administrador podrá prestar un libro a otra biblioteca, esto con la finalidad de satisfacer la demanda de libros de otras bibliotecas prestando nuestros libros. Lo hará con un formulario,introduciendo los datos: ISBN del libro y las biblioteca que solicita el préstamo
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Administrador}
		\UCitem{Propósito}{Prestar un libro a otra biblioteca, se llenara un formulario donde se introducirá la biblioteca a la que se le prestara el libro y el ISBN libro, se verificara si el libro se puede prestar, y si se tiene convenio con la biblioteca.}
		\UCitem{Entradas}{ISBN del libro y nombre del la biblioteca}
		\UCitem{Salidas}{Resitro en la base de datos junto con los mensajes .Mensaje \MSGref{MSG01}{Operación realizada exitosa mente}.Mensaje\MSGref{E3}{Datos introducidos incorrectamente.}Mensaje\MSGref{MSG03}{El libro no está disponible.}Mensaje\MSGref{MSG03}{La biblioteca no tiene convenio.}}
		\UCitem{Precondiciones}{ 
			\begin{itemize}
				\item El libro debe estar disponible para préstamo. 
				\item conexión a la base de datos.
				\item Se debe tener convenio con la biblioteca
			\end{itemize}
		}
		\UCitem{Postcondiciones}{El libro estará en la base de datos como prestado}
		\UCitem{Autor}{Rodriguez Ceravates Arturo y Monserrat Ceron Rodriguez .}
		\UCitem{Reviso}{Abidal y Diego}
	\end{UseCase}
		%-------------------------------------- COMIENZA descripción Trayectoria Principal
	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Presionara el botón:  \IUbutton{Préstamo interbibliotecario} de la pantalla \IUref{UI08.1}{Pantalla de inicio}.
		\UCpaso[\UCsist]Mostrara la pantalla \IUref{UI08.2}{pantalla de préstamo interbibliotecario}.
		\UCpaso[\UCactor]Ingresara los datos y presionara el botón: \IUbutton{Prestar} se verifican reglas de negocio  \BRref{RN08}{Estado de libro para préstamo} y \BRref{RN46}{Convenio con biblioteca}\Trayref{A} \Trayref{B}\Trayref{C}.
		\UCpaso[\UCsist] Mostrara el mensaje \MSGref{MSG01}{Operación realizada exitosa mente} Junto con el nombre el libro, la biblioteca, fecha actual y fecha de devolución
	\end{UCtrayectoria}
