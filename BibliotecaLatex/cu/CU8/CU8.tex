	\begin{UseCase}{CU08}{Préstamo interbibliotecario}{
		En este caso de uso tiene como objetivo prestar un libro a otra biblioteca
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Administrador}
		\UCitem{Propósito}{Prestar un libro de la biblioteca a otra biblioteca de acuerdo a la disponibilidad del libro. }
		\UCitem{Entradas}{Nombre del libro, nombre del la biblioteca, Fecha del día actual}
		\UCitem{Salidas}{una nota de préstamo donde se especificara el libro que se presta, la fecha y a quien se le presta}
		\UCitem{Precondiciones}{ 
			\begin{itemize}
				\item El libro debe estar disponible para préstamo. 
				\item conexión a la base de datos.
			\end{itemize}
		}
		\UCitem{Postcondiciones}{El libro estará prestado}
		\UCitem{Autor}{Rodriguez Ceravates Arturo y Monserrat Ceron Rodriguez .}
	\end{UseCase}
		%-------------------------------------- COMIENZA descripción Trayectoria Principal
	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Presionara el botón:  \IUbutton{Préstamo} de la pantalla \IUref{UI01}{Pantalla de inicio}.
		\UCpaso[\UCsist]Mostrara la pantalla \IUref{UI02{pantalla de préstamo}}.
		\UCpaso[\UCactor]Ingresara los datos y presionara el botón: \IUbutton{Prestar} \Trayref{A}.
		\UCpaso[\UCsist] Mostrara el mensaje \MSGref{MSG01}{Operación realizada exitosa mente}
	\end{UCtrayectoria}
			%-------------------------------------- COMIENZA descripción Trayectoria Alternativa.
		\begin{UCtrayectoriaA}{A}{Se ingreso mal alguno de los campos}
			\UCpaso[\UCsist] Mostrara el mensaje \MSGref{E3}{Datos introducidos incorrectamente.}
			\UCpaso[\UCsist] Termina el caso de uso.
		\end{UCtrayectoriaA}

%-------------------------------------		
		\begin{UCtrayectoriaA}{B}{El libro no esta disponible para el préstamo}
			\UCpaso[\UCsist] Mostrara el mensaje \MSGref{MSG03}{El libro no está disponible.}
			\UCpaso[\UCsist] Termina el caso de uso.
		\end{UCtrayectoriaA}
