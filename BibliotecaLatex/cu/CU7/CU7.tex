	\begin{UseCase}{CU07}{Prestamo de Libros a domicilio}{
		Este caso de uso tiene como objetivo que los clientes puedan pedir el prestamo de un libro fuera de la biblioteca.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Administrador}
		\UCitem{Propósito}{Se llenará un formulario donde se introducirá el identificador del usuario y el libro que se desea pedir a préstamo. El sistema verificará que el usuario esté debidamente registrado al sistema y que sea apto para pedir libros en préstamo y también se verificará si el libro es apto para ser prestado.}
		\UCitem{Entradas}{
						\begin{itemize}
							\item ID Usuario.
							\item ISBN del libro.
						\end{itemize}
					}
		\UCitem{Salidas}{Se redireccionará al usuario a una pantalla donde muestre los resultados de su peticion.}
		\UCitem{Precondiciones}{ 
			\begin{itemize}
				\item El administrador debe haber ingresado al sistema. 
				\item El usuario debe contar con un ID.
			\end{itemize}
		}
		\UCitem{Postcondiciones}{Se actualiza en la DB la lista de libros con los que el usuario cuenta en prestamo.}
		\UCitem{Autor}{Mora Guardado Daniel y Cortes Frias Diego Antonio.}
	\end{UseCase}
		%-------------------------------------- COMIENZA descripción Trayectoria Principal
	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Dará click en  \IUbutton{Pretsamo de Libros} de la interfaz \IUref{UI01}{Pantalla de Inicio}  .
		\UCpaso[\UCactor] Llenará el formulario de la interfaz \IUref{UI02}{Pantalla de prestamo a domicilio} \BRitem Se llenara con el ID del usuario y el 				ISBN del libro que el usuario quiere como pretsamo  \Trayref{A}.
		\UCpaso[\UCactor] Dará click en  \IUbutton{Realizar tramite} 
		\UCpaso[\UCsist] Se conecta a la BD \Trayref{B}.
		\UCpaso[\UCsist] Verificará la cuenta del usuario \Trayref{C} \Trayref{D} \Trayref{E} \Trayref{F}. .
		\UCpaso[\UCsist] Verificará la disponibilidad del libro \Trayref{G}.
		\UCpaso[\UCsist] Se muestra en \IUref{UI03}{Prestamo satisfacotrio} el mensaje  \MSGref{MSG01}{Fecha de devolucion del libro}
		\UCpaso[\UCactor] Dará click en  \IUbutton{Entendido} de  \IUref{UI03}{Prestamo satisfacotrio} y se redireccionara a  \IUref{UI01}{Pantalla de Inicio} .
	\end{UCtrayectoria}
